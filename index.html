⚙️ Configurar Supabase

Informe as credenciais do seu projeto Supabase para conectar o banco de dados.

Execute no SQL Editor do Supabase: -- 1. Tabela principal da adega CREATE TABLE IF NOT EXISTS adega ( id bigserial PRIMARY KEY, user_id uuid REFERENCES auth.users NOT NULL, nome text NOT NULL, uva text, local text, safra text, valor numeric, data date, quantidade integer DEFAULT 1, nota integer DEFAULT 0, avaliacao text, foto text, foto_hd text, criado_em timestamptz DEFAULT now() ); ALTER TABLE adega ENABLE ROW LEVEL SECURITY; -- Cada usuario ve e altera apenas seus proprios vinhos CREATE POLICY "select proprios" ON adega FOR SELECT USING (auth.uid() = user_id); CREATE POLICY "insert proprios" ON adega FOR INSERT WITH CHECK (auth.uid() = user_id); CREATE POLICY "update proprios" ON adega FOR UPDATE USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); CREATE POLICY "delete proprios" ON adega FOR DELETE USING (auth.uid() = user_id); -- 2. Tabela compartilhada de nomes de vinhos CREATE TABLE IF NOT EXISTS vinhos_nomes ( id bigserial PRIMARY KEY, nome text UNIQUE NOT NULL, uva text, safra text, contagem integer DEFAULT 1, criado_em timestamptz DEFAULT now() ); ALTER TABLE vinhos_nomes ENABLE ROW LEVEL SECURITY; CREATE POLICY "leitura autenticada" ON vinhos_nomes FOR SELECT USING (auth.role() = 'authenticated'); CREATE POLICY "insercao autenticada" ON vinhos_nomes FOR INSERT WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY "update autenticado" ON vinhos_nomes FOR UPDATE USING (auth.role() = 'authenticated'); -- 3. (Opcional) Funcao RPC para incremento atomico da contagem CREATE OR REPLACE FUNCTION upsert_vinho_nome(p_nome text, p_uva text, p_safra text) RETURNS void LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN INSERT INTO vinhos_nomes (nome, uva, safra, contagem) VALUES (p_nome, p_uva, p_safra, 1) ON CONFLICT (nome) DO UPDATE SET contagem = vinhos_nomes.contagem + 1, uva = COALESCE(EXCLUDED.uva, vinhos_nomes.uva), safra = COALESCE(EXCLUDED.safra, vinhos_nomes.safra); END; $$;
