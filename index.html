⚙️ Configurar Supabase

Informe as credenciais do seu projeto Supabase para conectar o banco de dados.

Execute no SQL Editor do Supabase: CREATE TABLE IF NOT EXISTS adega ( id bigserial PRIMARY KEY, user_id uuid REFERENCES auth.users NOT NULL, nome text NOT NULL, uva text, local text, safra text, valor numeric, data date, quantidade integer DEFAULT 1, nota integer DEFAULT 0, avaliacao text, foto text, foto_hd text, criado_em timestamptz DEFAULT now() ); ALTER TABLE adega ENABLE ROW LEVEL SECURITY; CREATE POLICY "usuarios proprios" ON adega USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); -- Tabela compartilhada de nomes de vinhos CREATE TABLE IF NOT EXISTS vinhos_nomes ( id bigserial PRIMARY KEY, nome text UNIQUE NOT NULL, uva text, safra text, contagem integer DEFAULT 1, criado_em timestamptz DEFAULT now() ); ALTER TABLE vinhos_nomes ENABLE ROW LEVEL SECURITY; CREATE POLICY "leitura publica" ON vinhos_nomes FOR SELECT USING (auth.role() = 'authenticated'); CREATE POLICY "insercao autenticada" ON vinhos_nomes FOR INSERT WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY "incremento autenticado" ON vinhos_nomes FOR UPDATE USING (auth.role() = 'authenticated');
